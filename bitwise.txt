import java.io.*;
class bitwise
{
   public static void main(String args[])
  {
   int x = 30;
   int y = 45;
   System.out.println(x&y);
   System.out.println(x|y);
   System.out.println(x^y);
   System.out.println(~x);
   System.out.println(x<<);
   System.out.println(x>>);


  }
}
